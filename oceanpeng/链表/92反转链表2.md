```c++
class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int m, int n) {
        ListNode* pre = nullptr;
        ListNode* now = head;
        ListNode* first = nullptr;
        ListNode* preFirst = nullptr;
        if(m == n) return head;
        int cnt = 0;
        while(cnt < n){
            cnt++;
            if(cnt == m-1){
                preFirst = now;
            }
            if(cnt >= m){
                if(cnt == m) first = now; 
                ListNode* tmp = now -> next;
                now -> next = pre;
                pre = now;
                now = tmp;
            }else{
                pre = now;
                now = now -> next;
            }
        }
        first -> next = now;
        if(preFirst != nullptr) preFirst -> next = pre;
        else head = pre;
        return head;
    }
};
```

要考虑第一个翻转的结点的前一个指针，然鹅前一个指针可能是空

所以要考虑到第一个就要翻转的情况。
